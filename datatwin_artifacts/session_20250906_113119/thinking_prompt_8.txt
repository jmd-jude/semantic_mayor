WITH segment_analysis AS (
    SELECT
        BUBBLES,
        AGE,
        HH_INCOME,
        COUNT(*) as attempts,
        SUM(CASE WHEN TARGET = 1 THEN 1 ELSE 0 END) as successes,
        COUNT(DISTINCT INDIVIDUALID) as unique_individuals,
        AVG(CONFIDENCE_LEVEL) as avg_confidence,
        COUNT(CREDIT_RATING) as credit_data_count,
        COUNT(PART_AFFILIATION) as political_data_count,
        COUNT(CASE WHEN PRES_KIDS = TRUE THEN 1 END) as kids_count
    FROM DM_CAMPAIGN
    WHERE AGE IS NOT NULL AND HH_INCOME IS NOT NULL
    GROUP BY BUBBLES, AGE, HH_INCOME
    HAVING attempts >= 50
),
segment_metrics AS (
    SELECT
        *,
        ROUND((successes::FLOAT / attempts) * 100, 2) as success_rate,
        ROUND((credit_data_count::FLOAT / attempts) * 100, 2) as credit_coverage,
        ROUND((political_data_count::FLOAT / attempts) * 100, 2) as political_coverage,
        ROUND((kids_count::FLOAT / attempts) * 100, 2) as kids_percentage
    FROM segment_analysis
)
SELECT *
FROM segment_metrics
WHERE success_rate > 0
ORDER BY success_rate DESC, attempts DESC
LIMIT 15
