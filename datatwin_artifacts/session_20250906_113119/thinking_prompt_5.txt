```sql
WITH contact_groups AS (
    SELECT 
      INDIVIDUALID,
      COUNT(*) as contact_count,
      SUM(CASE WHEN TARGET = 1 THEN 1 ELSE 0 END) as success_count
    FROM DM_CAMPAIGN
    GROUP BY INDIVIDUALID
  ),
  demographic_coverage AS (
    SELECT
      contact_count,
      success_count > 0 as is_success,
      COUNT(*) as group_size,
      AVG(CASE WHEN d.GENDER IS NOT NULL THEN 1 ELSE 0 END) as gender_coverage,
      AVG(CASE WHEN d.AGE IS NOT NULL THEN 1 ELSE 0 END) as age_coverage,
      AVG(CASE WHEN d.HH_INCOME IS NOT NULL THEN 1 ELSE 0 END) as income_coverage,
      AVG(CASE WHEN d.CREDIT_RATING IS NOT NULL THEN 1 ELSE 0 END) as credit_coverage,
      AVG(CASE WHEN d.PART_AFFILIATION IS NOT NULL THEN 1 ELSE 0 END) as political_coverage,
      AVG(CONFIDENCE_LEVEL) as avg_confidence,
      AVG(CASE WHEN d.PRES_KIDS IS NOT NULL THEN 1 ELSE 0 END) as kids_coverage
    FROM contact_groups cg
    JOIN DM_CAMPAIGN d ON cg.INDIVIDUALID = d.INDIVIDUALID
    GROUP BY contact_count, success_count > 0
  )
  SELECT 
    contact_count,
    is_success,
    group_size,
    ROUND(100 * gender_coverage, 2) as pct_gender_coverage,
    ROUND(100 * age_coverage, 2) as pct_age_coverage,
    ROUND(100 * income_coverage, 2) as pct_income_coverage,
    ROUND(100 * credit_coverage, 2) as pct_credit_coverage,
    ROUND(100 * political_coverage, 2) as pct_political_coverage,
    ROUND(100 * kids_coverage, 2) as pct_kids_coverage,
    ROUND(avg_confidence, 3) as avg_confidence
  FROM demographic_coverage
  ORDER BY contact_count, is_success
```
