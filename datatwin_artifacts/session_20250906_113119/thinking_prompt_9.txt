```sql
WITH temporal_success AS (
    SELECT
        BUBBLES,
        PURCHASE_DATE_RANGE,
        COUNT(*) as attempts,
        COUNT(DISTINCT INDIVIDUALID) as unique_individuals,
        SUM(CASE WHEN TARGET = 1 THEN 1 ELSE 0 END) as successes,
        SUM(CASE WHEN PRES_KIDS = TRUE THEN 1 ELSE 0 END) as with_kids,
        AVG(CASE WHEN CREDIT_RATING IS NOT NULL OR PART_AFFILIATION IS NOT NULL THEN 1 ELSE 0 END) as enriched_data_ratio
    FROM DM_CAMPAIGN
    WHERE BUBBLES IN ('Working for the Weekend', 'Date Nights City Lights', 'Urban Elites', 'Social Climbers')
    GROUP BY BUBBLES, PURCHASE_DATE_RANGE
    HAVING attempts >= 10
),
trend_analysis AS (
    SELECT
        BUBBLES,
        PURCHASE_DATE_RANGE,
        attempts,
        unique_individuals,
        successes,
        ROUND((successes::FLOAT / attempts) * 100, 2) as success_rate,
        ROUND((with_kids::FLOAT / attempts) * 100, 2) as kids_percentage,
        ROUND(enriched_data_ratio * 100, 2) as enriched_data_pct,
        LAG(successes::FLOAT / attempts) OVER (PARTITION BY BUBBLES ORDER BY PURCHASE_DATE_RANGE) as prev_success_rate
    FROM temporal_success
)
SELECT
    BUBBLES,
    MIN(PURCHASE_DATE_RANGE) as first_date,
    MAX(PURCHASE_DATE_RANGE) as last_date,
    COUNT(*) as time_periods,
    AVG(success_rate) as avg_success_rate,
    STDDEV(success_rate) as success_rate_volatility,
    CORR(success_rate, kids_percentage) as kids_correlation,
    CORR(success_rate, enriched_data_pct) as data_quality_correlation,
    AVG(CASE WHEN success_rate > COALESCE(prev_success_rate, 0) THEN 1 ELSE 0 END) as improvement_ratio
FROM trend_analysis
GROUP BY BUBBLES
ORDER BY avg_success_rate DESC
```
